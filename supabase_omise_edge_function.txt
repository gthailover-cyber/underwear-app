
// This is a template for a Supabase Edge Function to handle Omise Charges
// You can deploy this using 'supabase functions deploy process-omise-payment'

import { serve } from "https://deno.land/std@0.168.0/http/server.ts"
import { createClient } from 'https://esm.sh/@supabase/supabase-js@2.7.1'

// Omise SDK for Deno/Node
// Note: You might need to use fetch directly or a compatible library

const OMISE_SECRET_KEY = Deno.env.get('OMISE_SECRET_KEY')
const SUPABASE_URL = Deno.env.get('SUPABASE_URL')
const SUPABASE_SERVICE_ROLE_KEY = Deno.env.get('SUPABASE_SERVICE_ROLE_KEY')

serve(async (req) => {
  try {
    const { token, amount, userId, currency = 'thb' } = await req.json()

    // 1. Create Charge with Omise
    const chargeResponse = await fetch('https://api.omise.co/charges', {
      method: 'POST',
      headers: {
        'Authorization': `Basic ${btoa(OMISE_SECRET_KEY + ':')}`,
        'Content-Type': 'application/json'
      },
      body: JSON.stringify({
        amount: amount * 100, // Omise uses satangs
        currency: currency,
        card: token.startsWith('tokn_') ? token : undefined,
        source: token.startsWith('src_') ? token : undefined,
        return_uri: 'https://your-app.com/wallet-callback'
      })
    })

    const charge = await chargeResponse.json()

    if (charge.status === 'failed') {
      throw new Error(charge.failure_message)
    }

    // 2. If Success, update user balance in Supabase
    if (charge.status === 'successful') {
      const supabase = createClient(SUPABASE_URL, SUPABASE_SERVICE_ROLE_KEY)
      
      const { error: rpcError } = await supabase.rpc('add_coins', {
        user_id: userId,
        amount: amount
      })

      if (rpcError) throw rpcError
      
      return new Response(JSON.stringify({ success: true, charge_id: charge.id }), {
        headers: { 'Content-Type': 'application/json' },
        status: 200
      })
    }

    // 3. Handle Pending (for PromptPay/Redirect)
    return new Response(JSON.stringify({ 
      success: true, 
      status: 'pending', 
      authorize_uri: charge.authorize_uri 
    }), {
      headers: { 'Content-Type': 'application/json' },
      status: 200
    })

  } catch (error) {
    return new Response(JSON.stringify({ error: error.message }), {
      headers: { 'Content-Type': 'application/json' },
      status: 400
    })
  }
})
